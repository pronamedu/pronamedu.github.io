Age of ed, y

x + y = 74 ------ x = 74 - y

x - 3 = 3(y - 3)
(74 - y) - 3 = 3y - 9
74 - y - 3 = 3y - 9
- y - 3y = - 9 - 71
- 4y = - 80
4y = 80
y = 20
x = 74 - y
x = 74 - 20
x = 54
